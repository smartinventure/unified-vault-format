### **2. Build Scripts**

#### **build-native.ps1**
```powershell
#!/usr/bin/env pwsh
# Build all platform-specific native binaries

param(
    [string]$Configuration = "Release"
)

$ErrorActionPreference = "Stop"

# Platform targets
$platforms = @(
    "win-x64",
    "win-x86", 
    "linux-x64",
    "linux-arm64",
    "osx-x64",
    "osx-arm64"
)

# Clean previous builds
Write-Host "🧹 Cleaning previous builds..." -ForegroundColor Yellow
Remove-Item -Path "dist" -Recurse -Force -ErrorAction SilentlyContinue
New-Item -Path "dist/native" -ItemType Directory -Force | Out-Null

# Build for each platform
foreach ($platform in $platforms) {
    Write-Host "🔨 Building for $platform..." -ForegroundColor Green
    
    $outputPath = "dist/native/$platform"
    
    dotnet publish UvfLib.Native `
        --configuration $Configuration `
        --runtime $platform `
        --self-contained true `
        --output $outputPath `
        /p:PublishAot=true `
        /p:PublishTrimmed=true `
        /p:InvariantGlobalization=true
    
    if ($LASTEXITCODE -ne 0) {
        Write-Error "❌ Build failed for $platform"
        exit 1
    }
    
    Write-Host "✅ $platform build complete" -ForegroundColor Green
}

Write-Host "🎉 All native binaries built successfully!" -ForegroundColor Cyan
Write-Host "📁 Output location: dist/native/" -ForegroundColor Cyan
```

#### **Python Example**
```python
# Bindings/Python/uvf_vault/vault.py
import ctypes
import os
import platform

def _get_native_lib_path():
    """Get path to native library based on current platform"""
    system = platform.system().lower()
    machine = platform.machine().lower()
    
    if system == "windows":
        if machine in ["amd64", "x86_64"]:
            return "../../../dist/native/win-x64/uvflib_native.dll"
        else:
            return "../../../dist/native/win-x86/uvflib_native.dll"
    elif system == "linux":
        if machine in ["aarch64", "arm64"]:
            return "../../../dist/native/linux-arm64/uvflib_native.so"
        else:
            return "../../../dist/native/linux-x64/uvflib_native.so"
    elif system == "darwin":
        if machine in ["arm64"]:
            return "../../../dist/native/osx-arm64/uvflib_native.dylib"
        else:
            return "../../../dist/native/osx-x64/uvflib_native.dylib"
    
    raise RuntimeError(f"Unsupported platform: {system}-{machine}")

class UvfVault:
    def __init__(self):
        lib_path = _get_native_lib_path()
        self.lib = ctypes.CDLL(lib_path)
        self._setup_function_signatures()
```
