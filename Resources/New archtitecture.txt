## 🔧 **How UVF Decorators Work**

### **UvfDecorator Implementation Example:**

```csharp
// In UvfLib.Storage NuGet package
using StorageLib.Abstractions;
using UvfLib.Core;

namespace UvfLib.Storage.Decorators
{
    public class UvfDecorator : IStorage, IDisposable
    {
        private readonly IStorage _underlyingStorage;
        private readonly UvfCrypto _crypto;
        private readonly string _vaultFile;

        public UvfDecorator(IStorage underlyingStorage, string password, string vaultFile)
        {
            _underlyingStorage = underlyingStorage ?? throw new ArgumentNullException(nameof(underlyingStorage));
            _crypto = new UvfCrypto(password);
            _vaultFile = vaultFile;
        }

        // Implement IStorage interface - all methods decrypt/encrypt transparently
        public async Task<IntPtr> OpenAsync(string path, int flags, CancellationToken cancellationToken = default)
        {
            // 1. Map virtual path to encrypted file path in vault
            string encryptedPath = MapToEncryptedPath(path);
            
            // 2. Open the encrypted file from underlying storage
            IntPtr encryptedHandle = await _underlyingStorage.OpenAsync(encryptedPath, flags, cancellationToken);
            
            // 3. Create decrypted stream wrapper
            var decryptedHandle = new UvfFileHandle(encryptedHandle, _crypto, _underlyingStorage);
            
            return decryptedHandle.CreateContext();
        }

        public async Task ReadAsync(IntPtr fileHandle, long offset, long size, IntPtr buffer, CancellationToken cancellationToken = default)
        {
            var uvfHandle = UvfFileHandle.FromContext(fileHandle);
            
            // Read encrypted data from underlying storage
            byte[] encryptedData = new byte[size];
            IntPtr encryptedBuffer = Marshal.AllocHGlobal((int)size);
            try
            {
                await uvfHandle.UnderlyingStorage.ReadAsync(uvfHandle.UnderlyingHandle, offset, size, encryptedBuffer, cancellationToken);
                Marshal.Copy(encryptedBuffer, encryptedData, 0, (int)size);
                
                // Decrypt the data
                byte[] decryptedData = _crypto.Decrypt(encryptedData, offset);
                
                // Copy decrypted data to output buffer
                Marshal.Copy(decryptedData, 0, buffer, decryptedData.Length);
            }
            finally
            {
                Marshal.FreeHGlobal(encryptedBuffer);
            }
        }

        public async Task WriteAsync(IntPtr fileHandle, long offset, long size, IntPtr buffer, CancellationToken cancellationToken = default)
        {
            var uvfHandle = UvfFileHandle.FromContext(fileHandle);
            
            // Get plain data from input buffer
            byte[] plainData = new byte[size];
            Marshal.Copy(buffer, plainData, 0, (int)size);
            
            // Encrypt the data
            byte[] encryptedData = _crypto.Encrypt(plainData, offset);
            
            // Write encrypted data to underlying storage
            IntPtr encryptedBuffer = Marshal.AllocHGlobal(encryptedData.Length);
            try
            {
                Marshal.Copy(encryptedData, 0, encryptedBuffer, encryptedData.Length);
                await uvfHandle.UnderlyingStorage.WriteAsync(uvfHandle.UnderlyingHandle, offset, encryptedData.Length, encryptedBuffer, cancellationToken);
            }
            finally
            {
                Marshal.FreeHGlobal(encryptedBuffer);
            }
        }

        // All other IStorage methods delegate to underlying storage with encryption/decryption
        public async Task<bool> FileExistsAsync(string filePath, CancellationToken cancellationToken = default)
        {
            string encryptedPath = MapToEncryptedPath(filePath);
            return await _underlyingStorage.FileExistsAsync(encryptedPath, cancellationToken);
        }

        // ... (all other IStorage methods implemented similarly)

        private string MapToEncryptedPath(string virtualPath)
        {
            // Map "/documents/secret.txt" to vault internal path like "/vault/38f7a2b9c1d4e5f6.uvf"
            return _crypto.GetEncryptedPath(virtualPath);
        }

        public void Dispose()
        {
            _crypto?.Dispose();
            _underlyingStorage?.Dispose();
        }
    }
}
```

        public async Task<Stream> GetFileAsync(string virtualPath)
        {
            return await _vault.OpenReadAsync(virtualPath);
        }
        
        public async Task<IEnumerable<FileEntry>> ListDirectoryAsync(string virtualPath)
        {
            var vaultEntries = await _vault.ListDirectoryAsync(virtualPath);
            return vaultEntries.Select(ve => new FileEntry 
            { 
                Name = ve.Name, 
                IsDirectory = ve.IsDirectory 
            });
        }
    }
}
