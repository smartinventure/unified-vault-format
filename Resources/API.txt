## 🏗️ **API Design Concepts**

### **Option 1: Unified Facade Pattern**
```csharp
// Single entry point, format abstracted away
IVaultFileSystem vault = VaultFileSystem.Open(config);
```

**Pros**: Simple, consistent interface regardless of format
**Cons**: Might hide format-specific optimizations

### **Option 2: Fluent Builder Pattern** ⭐ **RECOMMENDED**
```csharp
var vault = VaultFileSystem.Builder()
    .WithPassword("secure-password")
    .WithFormat(VaultFormat.UVF)
    .WithEncryptionMode(EncryptionMode.FullEncryption)
    .WithLocation(@"C:\MyVault")
    .WithCaching(enabled: true)
    .Initialize(); // or .Open() for existing
```

**Pros**: Discoverable, flexible, type-safe configuration
**Cons**: Slightly more verbose

### **Option 3: Factory with Configuration Object**
```csharp
var config = new VaultConfiguration {
    Password = "secure-password",
    Format = VaultFormat.CryptomatorV8,
    Location = @"C:\MyVault",
    EncryptionMode = EncryptionMode.ContentOnly
};
var vault = VaultFileSystem.Create(config);
```

### **2. Fluent Builder Pattern Implementation**

```
UvfLib.FileSystem/Builder/
├── VaultFileSystemBuilder.cs       # Main fluent builder
├── VaultConfigurationBuilder.cs    # Configuration-specific builder  
└── BuilderExtensions.cs            # Extension methods
```

```csharp
// UvfLib.FileSystem/Builder/VaultFileSystemBuilder.cs
public class VaultFileSystemBuilder
{
    private VaultConfiguration _config = new();
    
    public VaultFileSystemBuilder WithPassword(string password)
    {
        _config.Password = password;
        return this;
    }
    
    public VaultFileSystemBuilder WithFormat(VaultFormat format)
    {
        _config.Format = format;
        return this;
    }
    
    public VaultFileSystemBuilder WithEncryptionMode(EncryptionMode mode)
    {
        _config.EncryptionMode = mode;
        return this;
    }
    
    public VaultFileSystemBuilder WithLocation(string path)
    {
        _config.VaultPath = path;
        return this;
    }
    
    // FACTORY METHOD - Creates appropriate adapter
    public async Task<IVaultFileSystem> InitializeAsync()
    {
        ValidateConfiguration();
        
        // Create core vault using open-source library
        var coreVault = await CreateCoreVaultAsync();
        
        // ADAPTER PATTERN - Return format-specific adapter
        return _config.Format switch
        {
            VaultFormat.UVF => new UvfVaultAdapter(coreVault, _config),
            VaultFormat.CryptomatorV8 => new CryptomatorVaultAdapter(coreVault, _config),
            _ => throw new NotSupportedException($"Format {_config.Format} not supported")
        };
    }
    
    private async Task<UvfLib.Core.Vault> CreateCoreVaultAsync()
    {
        // Delegate to open-source core for actual vault creation
        return _config.Format switch
        {
            VaultFormat.UVF => UvfLib.Core.Vault.LoadUvfVault(
                File.ReadAllBytes(_config.VaultPath), _config.Password),
            VaultFormat.CryptomatorV8 => UvfLib.Core.Vault.LoadCryptomatorV8Vault(
                File.ReadAllBytes(_config.VaultPath), _config.Password),
            _ => throw new NotSupportedException()
        };
    }
}
```

### **3. Factory Pattern Integration**

```csharp
// UvfLib.FileSystem/VaultFileSystem.cs (Main entry point)
public static class VaultFileSystem
{
    // FLUENT BUILDER entry point
    public static VaultFileSystemBuilder Builder() => new();
    
    // FACTORY METHOD (alternative entry point)
    public static async Task<IVaultFileSystem> CreateAsync(VaultConfiguration config)
    {
        return await Builder()
            .WithPassword(config.Password)
            .WithFormat(config.Format)
            .WithLocation(config.VaultPath)
            .WithEncryptionMode(config.EncryptionMode)
            .InitializeAsync();
    }
    
    // CONVENIENCE METHODS
    public static async Task<IVaultFileSystem> OpenExistingAsync(string path, string password)
    {
        var format = await DetectVaultFormatAsync(path);
        return await Builder()
            .WithPassword(password)
            .WithFormat(format)
            .WithLocation(path)
            .OpenAsync();
    }
}
```
